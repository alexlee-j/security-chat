setState(prev => ({ ...prev, prekeysUploaded: true }));

### 优化 2: 添加预密钥自动补充机制

**文件**: `apps/desktop/src/core/signal/key-management.ts`

**新增方法**:
```typescript
/**
 * 生成签名预密钥
 */
async generateSignedPrekeys(count: number): Promise<void> {
  await this.prekeys.generateSignedPrekeys(count);
}

/**
 * 获取预密钥状态
 */
async getPrekeysStatus(): Promise<{
  hasSignedPrekeys: boolean;
  hasOneTimePrekeys: boolean;
  signedPrekeysCount: number;
  oneTimePrekeysCount: number;
}> {
  const signedPrekeys = await this.getSignedPrekeys();
  const oneTimePrekeys = await this.getOneTimePrekeys();
  
  return {
    hasSignedPrekeys: signedPrekeys.length > 0,
    hasOneTimePrekeys: oneTimePrekeys.length > 0,
    signedPrekeysCount: signedPrekeys.length,
    oneTimePrekeysCount: oneTimePrekeys.length,
  };
}
```

### 优化 3: 优化登录时预密钥检查逻辑

**文件**: `apps/desktop/src/core/use-chat-client.ts`

**优化前**:
```typescript
async function onLogin(event: FormEvent<HTMLFormElement>): Promise<void> {
  event.preventDefault();

```

**优化后**:
```typescript
async function onLogin(event: FormEvent<HTMLFormElement>): Promise<void> {
  event.preventDefault();
  try {
    const result = await login(account, password);
    setAuthToken(result.accessToken);
    setAuth({ token: result.accessToken, userId: result.userId });
    
    // 登录后初始化 Signal 协议并检查预密钥
    try {
      await signalActions.initialize();
      
      // 检查并补充预密钥
      const prekeysStatus = signalState.prekeysStatus;
      if (prekeysStatus && (!prekeysStatus.hasSignedPrekeys || 
          !prekeysStatus.hasOneTimePrekeys || 
          prekeysStatus.oneTimePrekeysCount < 100)) {
        console.log('[Login] Replenishing prekeys after login...');
        await signalActions.replenishPrekeys();
      }
    } catch (signalError) {
      console.error('[Login] Failed to initialize Signal protocol:', signalError);
      // Signal 初始化失败不影响登录
    }
    
    await Promise.all([loadConversations(), loadFriendData()]);
  } catch {
    showToast('登录失败，请检查账号密码或后端服务状态', 'error');
  }
}
```

## 📈 优化效果

### 优化前的问题

| 问题 | 影响 |
|------|------|
| 签名预密钥未存储在本地 | 无法验证签名预密钥的签名 |
| 预密钥用完后不会自动补充 | 可能导致无法建立新会话 |
| 登录时不检查预密钥状态 | 老用户可能缺少预密钥 |

### 优化后的改进

| 改进 | 效果 |
|------|------|
| 初始化时检查并生成预密钥 | ✅ 确保预密钥始终存在 |
| 预密钥不足时自动补充 | ✅ 保持 100 个一次性预密钥 |
| 登录后自动检查预密钥 | ✅ 老用户也能获得完整预密钥 |
| 状态中显示预密钥信息 | ✅ 可在 UI 中监控预密钥状态 |
