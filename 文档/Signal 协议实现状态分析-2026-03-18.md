## 一、核心问题清单

### 🔴 严重问题（阻塞 E2EE 功能）

| # | 问题 | 位置 | 影响 | 修复优先级 |
|---|------|------|------|------------|
| 1 | **注册时 Signal 密钥为空** | `apps/desktop/src/core/use-chat-client.ts` onRegister | 用户注册时没有生成 Signal 密钥，无法建立加密会话 | P0 |
| 2 | **预密钥上传被注释** | `apps/desktop/src/core/signal/message-encryption.ts` uploadPrekeys() | 客户端无法将预密钥上传到服务器，其他用户无法发起加密会话 | P0 |
| 3 | **消息发送未使用 Signal 加密** | `apps/desktop/src/core/api.ts` sendMessage() | 消息仅使用 Base64 编码，无端到端加密保护 | P0 |

### 🟡 中等问题（影响安全性）

| # | 问题 | 位置 | 影响 | 修复优先级 |
|---|------|------|------|------------|

| 8 | **localStorage 存储密钥** | `apps/desktop/src/core/signal/key-management.ts` LocalSecureStorage | 密钥以 JWK 明文存储，易受 XSS 攻击 | P2 |

#### 修复 1: 注册时生成 Signal 密钥

```typescript
// apps/desktop/src/core/use-chat-client.ts
onRegister: async (event) => {
  event.preventDefault();
  setAuthSubmitting(true);
  
  try {
    // 1. 生成 Signal 密钥对
    const signal = new SignalProtocol();
    const identityKeyPair = await signal.generateIdentityKeyPair();
    const signedPrekey = await signal.generateSignedPrekey(identityKeyPair, 1);
    
    // 2. 转换为 Base64
    const identityPublicKey = arrayBufferToBase64(identityKeyPair.publicKeyBytes);
    const signedPreKey = arrayBufferToBase64(
      await crypto.subtle.exportKey('raw', signedPrekey.keyPair.publicKey)
    );
    const signedPreKeySignature = arrayBufferToBase64(signedPrekey.signature);
    
    // 3. 保存密钥到本地存储
    await keyManager.saveKeyPair(identityKeyPair);
    await keyManager.saveSignedPrekeys(new Map([[1, signedPrekey]]));
    
    // 4. 注册
    const result = await register({
      username: registerEmail,
      email: registerEmail,
      phone: '',
      password,
      deviceName: navigator.userAgent,
      deviceType: getDeviceType(), // 根据 userAgent 判断
      identityPublicKey,
      signedPreKey,
      signedPreKeySignature,
    });
    
    // 5. 生成一次性预密钥并上传
    await keyManager.generateOneTimePrekeys(100);
    await messageEncryptionService.uploadPrekeys();
    
    // 6. 设置认证状态
    setAuth({ token: result.accessToken, userId: result.userId });
    showToast('注册成功', 'success');
  } catch (error) {
    console.error('Register failed:', error);
    showToast('注册失败：' + (error as Error).message, 'error');
  } finally {
    setAuthSubmitting(false);
  }
}
```

async uploadPrekeys(): Promise<void> {
  const signedPrekeys = await this.keyManager.getSignedPrekeys();
  const oneTimePrekeys = await this.keyManager.getOneTimePrekeys();
  
  if (signedPrekeys.length === 0) {
    throw new Error('No signed prekeys to upload');
  }
  
  // 获取当前设备 ID（从登录响应或本地存储）
  const deviceId = await this.getCurrentDeviceId();
  
  const data = {
    deviceId,
    prekeys: oneTimePrekeys.map(prekey => 
      this.uint8ArrayToBase64(
        new Uint8Array(
          await crypto.subtle.exportKey('raw', prekey.keyPair.publicKey)
        )
      )
    ),
  };
  
  // 取消注释，实际上传
  await api.uploadPrekeys(data);
}

async getPrekeyBundle(
  @CurrentUser() user: RequestUser,
  @Param('userId') userId: string,
  @Param('deviceId') deviceId: string,
) {
  // 验证好友关系或自己是所有者

    await this.assertIsFriend(user.userId, userId);
  }
  return this.userService.getPrekeyBundle(userId, deviceId);
}

// 在 FriendService 中添加
async areFriends(userId: string, otherUserId: string): Promise<boolean> {
  const [userToOther, otherToUser] = await Promise.all([
    this.friendshipRepository.findOne({
      where: { userId, friendId: otherUserId, status: 1 },
    }),
    this.friendshipRepository.findOne({
      where: { userId: otherUserId, friendId: userId, status: 1 },
    }),
  ]);
  return !!(userToOther && otherToUser);
}

#### 修复 5: 签名预密钥使用身份私钥签名

```typescript
// apps/desktop/src/core/signal/index.ts
static async generateSignedPrekey(
  identityKeyPair: IdentityKeyPair, 
  keyId: number
): Promise<SignedPrekey> {
  const keyPair = await crypto.subtle.generateKey(
    { name: 'ECDH', namedCurve: 'P-256' },
    true,
    ['deriveBits', 'deriveKey']
  );
  
  // 使用身份私钥对签名预密钥公钥进行签名
  const publicKeyBytes = await crypto.subtle.exportKey('raw', keyPair.publicKey);
  const signature = await crypto.subtle.sign(
    { name: 'ECDSA', hash: 'SHA-256' },
    identityKeyPair.privateKey,
    publicKeyBytes
  );
  
  return {
    keyId,
    keyPair,
    signature: new Uint8Array(signature),
  };
}
```

#### 修复 6: 完善 Double Ratchet 实现

```typescript
// apps/desktop/src/core/signal/index.ts
export class DoubleRatchet {
  /**
   * DH 棘轮 - 接收新的 DH 公钥时更新根密钥
   */
  static async dhRatchet(
    rootKey: Uint8Array,
    localKeyPair: CryptoKeyPair,
    remotePublicKey: Uint8Array
  ): Promise<{ rootKey: Uint8Array; chainKey: Uint8Array }> {
    // 计算 DH 共享密钥
    const sharedSecret = await X3DH.calculateSharedSecret(
      localKeyPair.privateKey,
      remotePublicKey
    );
    
    // 使用 HKDF 派生新的根密钥和链密钥
    const { rootKey: newRootKey, chainKey } = await this.hkdf(
      rootKey,
      sharedSecret,
      'DoubleRatchet'
    );
    
    return { rootKey: newRootKey, chainKey };
  }
  
  /**
   * HKDF 密钥派生
   */
  private static async hkdf(
    inputKeyMaterial: Uint8Array,
    salt: Uint8Array,
    info: string
  ): Promise<{ rootKey: Uint8Array; chainKey: Uint8Array }> {
    const encoder = new TextEncoder();
    const infoBytes = encoder.encode(info);
    
    // 使用 HMAC-SHA256 实现 HKDF
    const key = await crypto.subtle.importKey(
      'raw',
      salt,
      { name: 'HMAC', hash: 'SHA-256' },
      false,
      ['sign']
    );
    
    const prk = await crypto.subtle.sign('HMAC', key, inputKeyMaterial);
    
    const prkKey = await crypto.subtle.importKey(
      'raw',
      prk,
      { name: 'HMAC', hash: 'SHA-256' },
      false,
      ['sign']
    );
    
    const okm1 = await crypto.subtle.sign(
      'HMAC',
      prkKey,
      new Uint8Array([...infoBytes, 0x01])
    );
    
    const okm2 = await crypto.subtle.sign(
      'HMAC',
      prkKey,
      new Uint8Array([...new Uint8Array(okm1), ...infoBytes, 0x02])
    );
    
    return {
      rootKey: new Uint8Array(okm1),
      chainKey: new Uint8Array(okm2),
    };
  }
}
```
